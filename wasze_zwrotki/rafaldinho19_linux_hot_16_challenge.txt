# 1. ls
# Pokazuje nam zawartość katalogu

# 2. touch
# Tworzenie pliku np. 'touch new_file.txt'

# 3. pwd
# Wyświetlenie ścieżki, w której sie znajdujemy

# 4. cd
# Zmiana katalogu bieżącego, np. cd /home/user/Muzyka

# 5. mkdir
# Tworzenie katalogu, np.'mkdir new_folder'

# 6. rmdir
# Usunięcie katalogu, np 'rmdir new_folder'

# 7. man
# Wyświetlenie pomocy dla danej komendy np. 'man cd' pokaże opis komendy oraz sposób użycia

# 8. cat
# Łączenie plików, np. 'cat plik1.txt plik2.txt' lub 'cat plik1.txt plik2.txt > plik3.txt' z zapisem do nowego pliku

# 9. cp
# Kopiowanie pliku, np. cp sample.txt new_sample.txt

# 10. mv
# Przeniesienie pliku do innego katalogu np. 'mv sample.txt Documents', można zmienić tez nazwę pliku, np. 'mv sample.txt new_file.txt'

# 11. rm
# Usunięcie pliku np. 'rm sample.txt'

# 12. exit
# Zamykanie terminala

# 13. nano
# Niewielki terminalowy edytor tekstu

# 14. sudo 
# służy do uruchomienia aplikacji jako inny użytkownik (administrator)

# 15. find
# Wyszukiwania plików i folderów

# 16. clear
# Wyczyszczenie terminala

# 25. userdel

# 26. groupadd
# Utworzenie nowej grupy

# 27. groupdel
# Usunięcie grupy

# 28. usermod
# Dodawanie użytkowników do grup

# 29. sort
